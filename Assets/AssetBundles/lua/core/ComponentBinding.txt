local _M = {}           -- 局部的变量
_M._VERSION = '1.0'     -- 模块版本

local ComponentBinding = { __index = _M }

function _M:new(self)
    return setmetatable({ }, ComponentBinding)
end

function ComponentBinding:Add(self, nativeCompUserData, luaCompInstance)
	if (self.bindingDict == nil) then
		self.bindingDict = {}
	end

	if (self.bindingDict[nativeCompUserData] == nil) then
		self.bindingDict[nativeCompuserData] = luaCompInstance
	end
end

function ComponentBinding:GetLuaComponentInstance(self, nativeCompUserData)
	if (self.bindingDict == nil) then
		return nil
	end

	return self.bindingDict[nativeCompUserData]
end

--这里全局变量保存所有binding
__componentBinding = _M:new()

print("ComponentBinding completed.")

return _M
